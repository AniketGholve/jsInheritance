Q1 Create 2 objects - parent and child. Attach methods to parent and use those methods in child object using parents prototype

        function Human(name,age){
            this.name=name;
            this.age=age;
        }
        Human.prototype.getNameAndAge=function(){
            return `Name is : ${this.name} And Age is : ${this.age}`;
        }

        function Man(name,age){
            Human.call(this,name,age);
        }
        
        let HumanObject=Object.create(Human)
        Man.prototype=HumanObject.prototype;

        let data =new Man("Aniket",22);
        console.log(data.getNameAndAge());
        
       
       
        Q2 Write code to explain prototype chaining

        function Human(name,age){
            this.name=name;
            this.age=age;
        }
        Human.prototype.getNameAndAge=function(){
            return `Name is : ${this.name} And Age is : ${this.age}`;
        }

        function Man(name,age){
            Human.call(this,name,age);
        }
        
        let HumanObject=Object.create(Human)
        Man.prototype=HumanObject.prototype;

        let data =new Man("Aniket",22);
        console.log(data.getNameAndAge());

        the above code shows that Man has its own prototype as well as Human prototype this concept is known as prototype chaining 
        prototype chaining allow to inherite propert of object into another object
